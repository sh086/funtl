    # 商户层销售

    # 插入修改商户销售日志
insert into tc_operate_log(customer_code,no,type,tag_key,tag_name,apply_type,operate_id,original_value,target_value,notes,createtime,updatetime,operate_time)
SELECT merchant.merchant_code,merchant.merchant_serial_no,0 as type,41 as tag_key,'员工交接' as tag_name,
merchant.apply_type,'应业务要求后台批量修改' as operate_id,orignEmployee.username as original_value,
aaa.merchant_sale as target_value,'修改商户推荐销售' as notes,NOW() as operate_time,NOW() as createtime,now() as updatetime
from aaa aaa
LEFT JOIN tc_merchant merchant on merchant.merchant_code = aaa.merchant_code
LEFT JOIN tc_employee orignEmployee on orignEmployee.id = merchant.introducer_sale;
# 插入修改商户销售生效时间日志
insert into tc_operate_log(customer_code,no,type,tag_key,tag_name,apply_type,operate_id,original_value,target_value,notes,createtime,updatetime,operate_time)
SELECT merchant.merchant_code,merchant.merchant_serial_no,0 as type,41 as tag_key,'员工交接' as tag_name,
merchant.apply_type,'应业务要求后台批量修改' as operate_id,merchant.sale_effect_time as original_value,
'2021-06-01 00:00:00' as target_value,'修改商户推荐销售生效时间' as notes,NOW() as operate_time,NOW() as createtime,now() as updatetime
from aaa aaa
LEFT JOIN tc_merchant merchant on merchant.merchant_code = aaa.merchant_code;
# 更新语句
UPDATE tc_merchant merchant, aaa aaa,tc_employee employee
set merchant.introducer_sale = employee.id, merchant.sale_effect_time = '2021-06-01'
where merchant.merchant_code = aaa.merchant_code
and employee.username = aaa.merchant_sale;
